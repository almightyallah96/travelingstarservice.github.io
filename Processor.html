<script>

/* ====== CONFIG: paste your provider links / values here ====== */

const CONFIG = {

  PAYPAL_ME_USER: 'PAYPAL_ME_USERNAME',    // e.g. 'YourPayPalMe' (no slashes)

  STRIPE_LINK: 'STRIPE_LINK_HERE',         // full Stripe Payment Link URL

  SQUARE_LINK: 'SQUARE_LINK_HERE',         // full Square Payment Link URL

  CASHAPP_CASHTAG: 'CASHAPP_CASHTAG',      // e.g. '$YourCashtag' or 'YourCashtag'

  ZELLE_ADDR: 'ZELLE_EMAIL_OR_PHONE'       // e.g. your zelle-email@example.com or phone

};

/* =========================================================== */


document.getElementById('cashTag').value = CONFIG.CASHAPP_CASHTAG;

document.getElementById('zelleAddr').textContent = CONFIG.ZELLE_ADDR;


// Simple distance estimator (same approach as earlier — straight-line demo)

function estimateDistance(pickup, dropoff) {

  // If you want a real distance upgrade later we can connect to Google Maps API.

  // For now use a small heuristic or fallback random realistic range:

  return Math.max(1, Math.round((Math.random() * 20) + 2)); // 2–22 miles

}


function calculate() {

  const pickup = document.getElementById('pickup').value;

  const dropoff = document.getElementById('dropoff').value;

  const rate = parseFloat(document.getElementById('rate').value || '2.5');

  if(!pickup || !dropoff) { alert('Enter both pickup and drop-off'); return; }


  const dist = estimateDistance(pickup, dropoff);

  const total = (dist * rate).toFixed(2);


  document.getElementById('distanceOut').textContent = dist;

  document.getElementById('priceOut').textContent = total;

  document.getElementById('summary').style.display = 'block';

  document.getElementById('paymentOptions').style.display = 'grid';


  // store for use when opening payment links

  window.bookingData = { pickup, dropoff, rate, distance: dist, price: total };

}


document.getElementById('calcBtn').addEventListener('click', calculate);


// PayPal button: opens PayPal.Me link with amount (PayPal.Me supports /AMOUNT)

document.getElementById('paypalBtn').addEventListener('click', function(){

  const d = window.bookingData;

  if(!d){ alert('Calculate price first'); return; }

  const amt = encodeURIComponent(d.price);

  const user = CONFIG.PAYPAL_ME_USER;

  if(!user || user==='PAYPAL_ME_USERNAME'){ alert('Please set your PayPal.Me username in CONFIG'); return; }

  const url = 'https://paypal.me/' + user + '/' + amt;

  window.open(url, '_blank');

});


// Stripe button: open the configured payment link. Create payment links in Stripe Dashboard (paste URL in CONFIG.STRIPE_LINK)

document.getElementById('stripeBtn').addEventListener('click', function(){

  const url = CONFIG.STRIPE_LINK;

  if(!url || url==='STRIPE_LINK_HERE'){ alert('Please paste your Stripe Payment Link URL in CONFIG'); return; }

  // If you create many fixed-price links, you can optionally generate links per price in dashboard and paste here

  window.open(url, '_blank');

});


// Square button: open the configured Square payment link (paste the link in CONFIG.SQUARE_LINK)

document.getElementById('squareBtn').addEventListener('click', function(){

  const url = CONFIG.SQUARE_LINK;

  if(!url || url==='SQUARE_LINK_HERE'){ alert('Please paste your Square Payment Link URL in CONFIG'); return; }

  window.open(url, '_blank');

});


// Copy cash tag

document.getElementById('copyCash').addEventListener('click', function(){

  const tag = CONFIG.CASHAPP_CASHTAG;

  navigator.clipboard && navigator.clipboard.writeText(tag).then(function(){

    alert('Cashtag copied! Open Cash App and paste to pay.');

  }, function(){ alert('Copy failed — please select and copy manually: ' + tag); });

});
